Constructeurs :
a = {0, 0}
b = {8, 9}
c = {3, 3}

Accès aux composants :
b[0] = 8, b[1] = 9
c[0] = 3, c[1] = 3
c[0] = 7, c[1] = 3

Opérations avec scalaires :
a = {3, 3}
2 * a = {6, 6}
a * 2 = {6, 6}
b + 2 = {10, 11}
2 + b = {10, 11}

Opérations composées :
a += b += c : a = {10, 6}, b = {9, 5}
b = {9, 5}
c = {7, 3}
a = {5, 5}
a -= 2 : {3, 3}
a = {3, 3}
b = {9, 5}
a -= b : {-6, -2}

a = {-6, -2}
b = {9, 5}
c = {7, 3}
a = b * c + 2 : {65, 17}
a = 2 + b * c : {65, 17}

Incrémentation/Décrémentation :
a = {1, 1}
a++ = {1, 1}
a = {2, 2}
++a = {3, 3}
a-- = {3, 3}
a = {2, 2}
--a = {1, 1}

Comparaisons :
b == c : 1
b != c : 0
